import React, {useState} from 'react';

function ShowHide() {
    const [show, setShow] = useState(true);

    const handleClick = (event) => {
        setShow(!show);
    };


  return (
  <div>
    <button onClick={handleClick}>{show ? "Cotiza aqu√≠" : "En proceso..."}</button>
    {show && <h2>CONTACTENOS!</h2>}
  
    
    </div>
  );
}

export default ShowHide;
